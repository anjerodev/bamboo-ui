### Controllable

<PreviewContainer>

<ProgressCircles.Controllable slot="preview" client:load />

```tsx meta={fileName: ControllableProgressCircleDemo.tsx}
import { useState } from 'react'
import { css } from 'styled-system/css'
import { ProgressCircle } from '@/components/ui/progress-circle'

export function ControllableProgressCircleDemo() {
  const [value, setValue] = useState(0)

  const handleValueChange = (value: number) => {
    setValue((prev) => {
      const result = prev + value
      if (result > 100) return 100
      if (result < 0) return 0
      return result
    })
  }

  return (
    <div className={css({ display: 'flex', gap: 8, alignItems: 'center' })}>
      <Button onPress={() => handleValueChange(-25)}>- 25</Button>
      <ProgressCircle value={value} />
      <Button onPress={() => handleValueChange(25)}>+ 25</Button>
    </div>
  )
}
```

</PreviewContainer>
