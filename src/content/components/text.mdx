# Examples

### Paragraphs

<PreviewContainer>

<Texts.Default slot="preview" />

```tsx meta={fileName: TextDemo.tsx}
import { Text } from '@/components/ui/text'

export const TextDemo = () => {
  return (
    <div>
      <Text>Hello Bamboo UI</Text>
      <Text css={{ fontSize: 'xl', fontWeight: 'semibold' }}>
        Hello Bamboo UI
      </Text>
    </div>
  )
}
```
